void Delete()
{

	cout<<"Write the name of palyer who wants to quit.."<<endl;

	for( m=0;m<=lc;m++)
	{
		if(name[m]==del_name)
		{
			p=m;
			f=1;
			break;
		}
	}
	if(f==1)
	{
		cout<<name[p]<<" ,you have left the game"<<endl;
		for(m=p+1;m<=lc;m++)
		{
			name[m-1]=name[m];
			amount[m-1]=amount[m];
		}
		lc--;
	}
	else
	{
		cout<<"The player you entered is not playing"<<endl;
	}
	cout<<"\nEnter 0 to exit\n";
	cin>>call;
	if(call==0)
	game();
